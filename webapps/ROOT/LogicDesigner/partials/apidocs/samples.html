Here are some very basic samples of connecting to your Espresso Logic API for various languages.

These are only meant as a starting point: there are an infinite number ways to connect and utilize your API once it's up and running. 

$.ajax({
	type: 'GET',
	dataType: 'json',
	headers: {
		Accept: 'application/json',
		Authorization: 'Espresso {{data.selectedApiKey.apikey}}:1'},
	url: '{{data.serverUrl}}{{currentAccount.url_name}}/{{currentProject.url_name}}/{{data.selectedApiVersion.name}}/{{data.selectedResourceName}}',
	success: function(data) {
		for (var i = 0; i < data.length; i++) {
			console.log(data[i]);
		}
	},
	error: function(data) {
		console.log('Error code: ' + data.errorCode);
		console.log('Error message: ' + data.errorMessage);
	}
});

URL url = new URL("{{data.serverUrl}}{{currentAccount.url_name}}/{{currentProject.url_name}}/{{data.selectedApiVersion.name}}/{{data.selectedResourceName}}");
HttpURLConnection conn = (HttpURLConnection) url.openConnection();
conn.setRequestMethod("GET");
conn.setRequestProperty("Accept", "application/json");
conn.setRequestProperty("Authorization", "Espresso {{data.selectedApiKey.apikey}}:1");
 
BufferedReader br;
if (conn.getResponseCode() == 200)
	br = new BufferedReader(new InputStreamReader((conn.getInputStream())));
else
	br = new BufferedReader(new InputStreamReader((conn.getErrorStream())));
 
String output;
System.out.println("Output from server:\n");
while ((output = br.readLine()) != null) {
	System.out.println(output);
}

conn.disconnect();

$curl = curl_init();
curl_setopt($curl, CURLOPT_HTTPHEADER, array(
	'Accept: application/json',
	'Authorization: Espresso {{data.selectedApiKey.apikey}}:1'
));
curl_setopt($curl, CURLOPT_URL, 
	"{{data.serverUrl}}{{currentAccount.url_name}}/{{currentProject.url_name}}/{{data.selectedApiVersion.name}}/{{data.selectedResourceName}}");
curl_setopt($curl, CURLOPT_RETURNTRANSFER, TRUE);
$response = curl_exec($curl);
$json = json_decode($response);
var_dump($json);

import httplib
import json

conn = httplib.HTTPConnection("{{data.hostname}}")
headers = {'Accept': 'application/json', 'Authorization': 'Espresso {{data.selectedApiKey.apikey}}:1'}
conn.request('GET', '/rest/{{currentAccount.url_name}}/{{currentProject.url_name}}/{{data.selectedApiVersion.name}}/{{data.selectedResourceName}}', None, headers)
resp = conn.getresponse()
json = json.loads(resp.read())
if type(json) == dict and json.has_key('errorCode'):
	print "Error : " + json['errorMessage']
else:
	print "Success : " + str(json)

Installing our npm package is as simple as:

npm install espressologic

Once installed, using the SDK to begin making requests we point an instance of espressologic to our project and can immediately get started:

var espressologic, api, customers, newCustomer;

espressologic = require('espressologic');
api = espressologic.connect('https://{{data.hostname}}/rest/{{currentAccount.url_name}}/{{currentProject.url_name}}/{{data.selectedApiVersion.name}}', '{{data.selectedApiKey.apikey}}');

customers = api.endpoint('/demo:customers');
alphaCustomer = {
    name: "Alpha",
    credit_limit: "1234"
};

//PUT
customers.put(alphaCustomer, params).then(function (txSummary) {
    console.log(txSummary); //an array of objects updated
});

//GET
customers.get().then(function (data) {
    console.log(data); //an array which will now include customer: Alpha

    //objects returned include metadata specific to each record,
    //the most useful to us here is the endpoint href
    var alphaEndpoint = espressologic.endpoint(data[0]['@metadata'].href);

    //POST
        data[0].name = 'Alpha Updated';
        alphaEndpoint.post(data[0]).then(function(txSummary) {
            console.log(txSummary);
        });

    //DELETE
        alphaEndpoint.delete(data[0]).then(function(txSummary) {
            console.log(txSummary);
        });
});
